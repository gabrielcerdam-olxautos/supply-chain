sol

js
- "truffle-assertions": "^0.9.2"

- "@truffle/hdwallet-provider": "^2.0.0",

- "web3": "^1.7.0"

- "chai": "^4.3.6",

- "mocha": "^9.2.0",
    We use mocha to test our contracts
- "lite-server": "2.4.0",
